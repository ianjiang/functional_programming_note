var dispatch = (...args) => {
	let funs = _.toArray(args),
	    size = funs.length;

	return (...params) => {
		let ret = undefined,
			funIndex,
			fun;

		for (funIndex = 0; funIndex < size; funIndex++) {
			fun = funs[funIndex];
			ret = fun.apply(null, params);

			if (ret != null) return ret;
		}

		ret;
	};
};

var stringReverse = (s) => {
	return !_.isString(s) ? undefined : s.split('').reverse().join();
};

var rev = dispatch(stringReverse);
rev('abscd');

 var isa = (type, action) => {
 	return (obj) => {
 		if (obj.type === type) {
 			return action(obj)
 		}
 	}
 };
